# randomer.js

# select a random thing in an array
```js
    const randomer = require("randomer.js");

    var test = [
        "hello",
        "hi",
        "other stuff"
    ]

    console.log(randomer.array(test))
```

# random number

select a random number from 0 to the number selected

```js
    const randomer = require("randomer.js");

    console.log(randomer.number(100))

```
